Завдання 1. Огляд активних процесів
# 1. Виведення списку всіх активних процесів у системі (користувацькі + системні)
ps aux

# 2. Запуск інтерактивного монітора (після запуску натисніть 'M' для сортування за RAM)
top

# 3. Знаходження PID (Process ID) поточної оболонки (bash або zsh)
echo $$

Завдання 2. Робота у фоні та керування процесами
# 1. Запуск команди sleep на 1000 секунд у фоновому режимі (символ &)
sleep 1000 &

# 2. Перегляд списку активних фонових завдань у поточному терміналі
jobs

# 3. Повернення фонового процесу на передній план (fg - foreground)
# Якщо номер завдання 1, використовуємо %1
fg %1

# 4. Примусове завершення процесу. 
# Спочатку зупиняємо через Ctrl+Z, потім вбиваємо за PID (дізнатися PID можна через jobs -l)
kill -9 <PID>

# 5. Запуск команди, яка ігноруватиме сигнал виходу (SIGHUP) при закритті терміналу
nohup sleep 500 &

Завдання 3. Пріоритети та обмеження 
# 1. Запуск нової команди з низьким пріоритетом (nice 10)
# Значення nice варіюються від -20 (високий) до 19 (низький)
nice -n 10 sleep 300 &

# 2. Зміна пріоритету вже запущеного процесу (збільшуємо до 15)
# Потрібно вказати PID процесу, який ми запустили вище
renice +15 -p <PID>

# 3. Перегляд усіх лімітів ресурсів для поточного користувача
ulimit -a

Завдання 4. Моніторинг ресурсів
# 1. Перегляд вільного місця на дисках у зрозумілому форматі (Human-readable)
df -h

# 2. Перегляд статистики оперативної пам'яті (у мегабайтах/гігабайтах)
free -h
