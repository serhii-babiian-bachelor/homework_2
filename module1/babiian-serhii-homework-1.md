# Домашнє завдання №1: Основи Linux

## Завдання 1. Базові команди

```bash
# 1. Показуємо вміст домашнього каталогу
ls

# 2. Переходимо в Downloads та переглядаємо його вміст
cd ~/Downloads && ls

# 3. Створюємо порожній файл без використання touch (використовуємо перенаправлення потоку)
echo -n > myfile.txt

# 4. Переглядаємо вміст щойно створеного файлу
cat empty_file.txt

# 5. Перехід у домашній каталог абсолютним шляхом (замість 'user' твоє ім'я в системі)
cd /Users/user 

# 6. Перехід у домашній каталог відносним шляхом (тильда символізує home directory)
cd ~

## Завдання 2. Робота з документацією 
#1 Показує повну інструкцію до команди ls (список файлів). 
man ls	

#2 Виводить коротку довідку про те, як змінювати директорії.
help cd	

#3 Документація до команди cat (виведення вмісту файлів).
man cat	

#4 Це рекурсія! Інструкція про те, як користуватися самою інструкцією man.
man man	

#5 Швидка довідка для команди копіювання файлів (cp).
cp --help	

#6 Швидка довідка для команди переміщення або перейменування файлів (mv).
mv --help	

##Відповіді на питання:
#1 Який ключ ls показує приховані файли?
# Ключ -a (скорочення від all).

#2 Який ключ у cat нумерує рядки?
# Ключ -n (скорочення від number).

#3 Чим відрізняються man і --help?
# man відкриває повну електронну інструкцію в окремому інтерфейсі (пейджері), де можна гортати сторінки.
# --help виводить коротку довідку про доступні прапорці (флаги) прямо в потік термінала.

## Завдання 3. Міні-сценарій
# Сценарій автоматизації базових дій:

# Створюємо папку для роботи
mkdir test_folder

# Переходимо в створену папку
cd           

# Створюємо файл з текстом всередині
echo "Task 3 done" > report.txt  

# Переглядаємо всі файли (включаючи приховані) у списку
ls -la  

# Виходимо на рівень вище
cd ..                      

# Повертаємося в попередню директорію (test_folder)
cd -    

# Переглядаємо документацію команди echo
man echo
